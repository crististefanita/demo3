package com.endava.ai.atf.ui;

import com.endava.ai.atf.ui.support.ProtectedAccountSeed;
import com.endava.ai.core.reporting.StepLogger;
import com.endava.ai.ui.core.BaseTestUI;
import com.endava.ai.ui.factory.CustomerDataFactory;
import com.endava.ai.ui.model.CustomerData;
import com.endava.ai.ui.service.AccountService;
import com.endava.ai.ui.service.AuthenticationService;
import com.endava.ai.ui.validation.AccountValidation;
import com.endava.ai.ui.validation.AuthenticationValidation;
import org.testng.annotations.BeforeMethod;
import org.testng.annotations.Test;

public class AccountPasswordCredentialTransitionTests extends BaseTestUI {

    private final AuthenticationService authenticationService = new AuthenticationService();
    private final AccountService accountService = new AccountService();
    private final AuthenticationValidation authenticationValidation = new AuthenticationValidation();
    private final AccountValidation accountValidation = new AccountValidation();

    @BeforeMethod
    public void before() {
        authenticationService.clearSession();
        StepLogger.startStep("Password credential transition preconditions");
        StepLogger.pass("Ready");
    }

    @Test(description = "Positive: changed password rejects the old credential and reaches account overview with the new credential after session clear")
    public void changed_password_invalidates_the_old_credential_and_reaches_account_overview_with_the_new_credential() {
        CustomerData customer = ProtectedAccountSeed.mutationCustomer();
        String oldPassword = customer.getPassword();
        String newPassword = CustomerDataFactory.newStrongPassword();

        try {
            loginIntoProfile(customer, oldPassword);
            accountService.changePassword(oldPassword, newPassword);

            authenticationService.clearSession();
            authenticationService.openLogin();
            authenticationService.login(customer.getEmail(), oldPassword);
            authenticationValidation.assertLoginErrorContains("Invalid email or password");

            authenticationService.clearSession();
            loginIntoAccount(customer.getEmail(), newPassword);
        } finally {
            authenticationService.clearSession();
        }
    }

    @Test(description = "Negative: old credential can still reopen account overview after the claimed password change")
    public void old_credential_can_still_reopen_account_overview_after_the_claimed_password_change() {
        CustomerData customer = ProtectedAccountSeed.mutationCustomer();
        String oldPassword = customer.getPassword();
        String newPassword = CustomerDataFactory.newStrongPassword();

        try {
            loginIntoProfile(customer, oldPassword);
            accountService.changePassword(oldPassword, newPassword);

            authenticationService.clearSession();
            loginIntoAccount(customer.getEmail(), oldPassword);
        } finally {
            authenticationService.clearSession();
        }
    }

    @Test(description = "Positive: after old invalid and new valid credential transition, the same identity can reopen the protected profile surface")
    public void changed_password_can_turn_auth_surface_green_and_reopen_the_same_profile_surface() {
        CustomerData customer = ProtectedAccountSeed.mutationCustomer();
        String oldPassword = customer.getPassword();
        String newPassword = CustomerDataFactory.newStrongPassword();

        try {
            loginIntoProfile(customer, oldPassword);
            accountService.changePassword(oldPassword, newPassword);

            authenticationService.clearSession();
            authenticationService.openLogin();
            authenticationService.login(customer.getEmail(), oldPassword);
            authenticationValidation.assertLoginErrorContains("Invalid email or password");

            authenticationService.clearSession();
            loginIntoAccount(customer.getEmail(), newPassword);

            accountService.openProfile();
            accountValidation.assertProfileMatches(customer);
        } finally {
            authenticationService.clearSession();
        }
    }

    @Test(description = "Negative: both the old and the new credential can reopen the same protected identity after the claimed password change")
    public void both_old_and_new_credentials_can_reopen_the_same_protected_identity_after_the_claimed_change() {
        CustomerData customer = ProtectedAccountSeed.mutationCustomer();
        String oldPassword = customer.getPassword();
        String newPassword = CustomerDataFactory.newStrongPassword();

        try {
            loginIntoProfile(customer, oldPassword);
            accountService.changePassword(oldPassword, newPassword);

            authenticationService.clearSession();
            loginIntoAccount(customer.getEmail(), oldPassword);
            accountService.openProfile();
            accountValidation.assertProfileMatches(customer);

            authenticationService.clearSession();
            loginIntoAccount(customer.getEmail(), newPassword);
            accountService.openProfile();
            accountValidation.assertProfileMatches(customer);
        } finally {
            authenticationService.clearSession();
        }
    }

    private void loginIntoProfile(CustomerData customer, String password) {
        loginIntoAccount(customer.getEmail(), password);
        accountService.openProfile();
        accountValidation.assertOnProfile();
    }

    private void loginIntoAccount(String email, String password) {
        authenticationService.openLogin();
        authenticationService.login(email, password);
        authenticationValidation.assertOnAccountOverview();
    }
}
